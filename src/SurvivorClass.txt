Actor FlashLight : Weapon
{
	+WEAPON.CHEATNOTWEAPON
	+WEAPON.NOALERT
	States
	{
		Spawn:
			TNT1 A 1
			stop
		Select:
			ZZFF A 1 A_Raise
			loop
		Deselect:
			ZZFF A 1 A_Lower
			loop
		Ready:
			ZZFF A 0 A_FireCustomMissile("FlashLightLight")
			ZZFF A 1 A_FireCustomMissile("FlashLightLight")
			loop
		Fire:
			ZZFF A 0 A_FireCustomMissile("FlashLightLight")
			ZZFF A 1 A_FireCustomMissile("FlashLightLight")
			Goto Ready
	}
}

actor Knife : Weapon
{
	Weapon.Kickback 0
	+WEAPON.MELEEWEAPON
	+WEAPON.CHEATNOTWEAPON
	+WEAPON.NOALERT
	+NOEXTREMEDEATH
	Obituary "%o has been hunted by %k"

	States
	{
		Spawn:
			KNFE A -1
			loop
		Ready:
			KNFG A 1 A_WeaponReady
			loop
		Deselect:
			KNFG A 1 A_Lower
			loop
		Select: 
			KNFG A 1 A_Raise
			loop
		Fire:
			KNFG B 2
			KNFG C 2 A_CustomPunch ((200),0,0,"KnifePuff")
			KNFG DEFED 2
			KNFG A 2 A_ReFire
			goto Ready
	}
}
